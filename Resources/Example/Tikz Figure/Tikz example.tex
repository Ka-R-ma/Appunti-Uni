\begin{itemize}
    \item Simple Tikz figure:
          \begin{figure}[!h]
              \centering
              \begin{tikzpicture}[scale = 1, every node/.style={scale = 0.75}]
                  \node (a) at (0, 0) {a};
                  \node (b) at (2, 0) {b};

                  \draw (a) to (b);
              \end{tikzpicture}
          \end{figure}

    \item Tree figure
          \begin{figure}[!h]
              \centering
              \begin{forest}
                  for tree = {s sep = 15}
                  [a,
                  [b]
                      [c]
                  ]
              \end{forest}
          \end{figure}
    \item Tab figure
          \begin{figure}[!h]
              \centering
              \begin{tabular}{|c||c|}
                \hline
                Coln1 & Coln2 \\
                \hline
              \end{tabular}
          \end{figure}

    \item Automaton figure
    \begin{figure}[!h]
        \centering
        \begin{tikzpicture}[node distance = 2cm, on grid, auto, every node/.style={scale=0.85}, initial text = start]
        
            \tikzstyle{every state} = [fill={rgb:black,1;white,10}]
        
            \node (q0) [state, initial] {\(q_{0}\)};

            \node (q1) [state, right of = q0] {\(q_{1}\)};

            \path [->] (q0) edge node {1} (q1);
            
        \end{tikzpicture}
    \end{figure}
\end{itemize}
